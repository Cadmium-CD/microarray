//reset function
void AD7606_Reset(){
        cs = 1;
        
        rst = 0;
       
        rst = 1;
        rst = 1;
        rst = 1;
      
        rst = 0;
}


 
void printSamples()
{   ///////

   ///////

} 
void record(){
        cs = 0;
        sample1[wp] = spi.write(0x0000);
        sample2[wp] = spi.write(0x0000);
        sample3[wp] = spi.write(0x0000);
        cs = 1;
        wp++;
 }

int main() {
    total.start();
    spi.format(16,2);
    spi.frequency(50000000);
    convst.period(1.0f); 
    convst.write(1.0f);
    AD7606_Reset();
    while(abs(r13 - r12 - r23) > 2){
        convst.period_us(INTERAL_US);
        convst.write(0.5f);
   
         t.start(); 
         while((wp < BUF_LEN)){
             button.fall(&record); 
          }
  // myled = 0;
         s = t.read();
         t.stop();
         convst.write(1.0f);   
         wp = 0;
         scalling();
         r12 = xcorr(sample1_s, sample2_s);
         r13 = xcorr(sample1_s, sample3_s);
         r23 = xcorr(sample2_s, sample3_s);
         myled2 = 0;
   } 
    
    //printSamples();
    printf("Running time = %f\n",s);
    printf("Total time = %f\n",total.read());
    printf("r12 = %d\nr13 = %d\nr23 = %d\n",r12,r13,r23);
    printf("Angle = %f\n",cal_angle());
    
       
}
